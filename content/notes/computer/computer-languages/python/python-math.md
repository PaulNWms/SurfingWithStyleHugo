```python
42 ** 100
57 % 2  # modulus (remainder)
1 - (2+4j)

# Integers are Objects!
print(dir(42))

help((42).bit_length)

# floor division
a // b

# exponentiation
a ** b

# bit SHIFT left, right
a << b; a >> b

# bitwise NOT
~a

# output binary
bin(10)

# Boolean operators: and, or, not

# boolean XOR
(x > 1) != (x < 10)

# identity operators
a is b
a is not b

# membership operators
a in b
a not in b

# complex numbers
x = 1 + 2j
```
